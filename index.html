// Security Middleware
export const authGuard = async (req) => {
  const token = req.headers.get("auth");
  
  try {
    const decoded = await verify(token);
    return decoded.isValid ? 
      next() : unauthorized();
  } catch (err) {
    return error("Secure flow failed");
  }
}
